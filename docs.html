// extract token immediately on load

if (token) {

    // MUST remove from URL for security
    window.history.replaceState(null, null, window.location.pathname);

  • token source: window.location.hash

  • risk: URL leak if not cleared
  • edge case: token missing → redirect to login

3. Verify

# send token to backend for EVERY protected request
GET https://ageinx.vercel.app/auth/userinfo

  • method: GET
  • header: Authorization: Bearer <token>

  • invalid token: 401
  • expired token: 401
  • network fail: treat as 503
  • MUST: verify on EVERY protected request
  • NEVER: trust frontend-only auth

React / Next.js

  • runs: on mount (useEffect)
  • action: extracts token once
  • storage: stores token client-side
  • success: redirects after success
  • fallback: redirect to /login
  • compatibility: no SSR dependency required (works in SPA + Next.js)
import { useEffect } from 'react';
import { useRouter } from 'next/router';

export default function AuthCallback() {
    const router = useRouter();

    useEffect(() => {
        // read and store token instantly on mount
        const token = new URLSearchParams(window.location.hash.substring(1)).get('token');
        if (token) {
            localStorage.setItem('ax_token', token);
            router.push('/dashboard'); 
        } else {
            router.push('/login');
        }
    }, []);

    return <div>Logging in...</div>;
}

Node.js / Express

  • pattern: middleware-based protection
  • execution: runs before route handler
  • action: extracts token from headers, calls Ageinx API
  • success: attaches req.user
  • failure: blocks request if invalid
  • security: fail closed (secure by default)
  • optional: cache userinfo (Redis 5–10 min)
const axios = require('axios');

async function requireAuth(req, res, next) {
    // extract token from headers
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: "Denied" });

    try {
        // call Ageinx API to attach req.user
        const { data } = await axios.get('https://ageinx.vercel.app/auth/userinfo', {
            headers: { 'Authorization': `Bearer ${token}` }
        });
        req.user = data; 
        next();
    } catch (error) {
        // fail closed (secure by default)
        return res.status(401).json({ error: "Invalid token" });
    }
}

Python / FastAPI

  • pattern: dependency injection (FastAPI Depends)
  • network: async HTTP call (httpx)
  • handling: automatic 401 handling
  • DX: clean route-level protection, reusable across endpoints
  • perf: minimal overhead
import httpx
from fastapi import FastAPI, Depends, HTTPException
from fastapi.security import HTTPBearer

app = FastAPI()
security = HTTPBearer()

async def verify_user(credentials = Depends(security)):
    async with httpx.AsyncClient() as client:
        # verify token async
        response = await client.get(
            "https://ageinx.vercel.app/auth/userinfo",
            headers={"Authorization": f"Bearer {credentials.credentials}"}
        )
        if response.status_code != 200:
            raise HTTPException(status_code=401, detail="Invalid token")
        return response.json()

Security

  • hashing: bcrypt / argon2
  • tokens: signed JWT (HS256/RS256)
  • storage: no plaintext passwords
  • isolation: app never sees credentials
  • replay attacks: mitigated server-side
  • rate limiting: built-in (no need to implement)
  • brute force: blocked automatically
  • XSS risk: localStorage unsafe (prod warning)

Common Mistakes

  • storing token forever: bad (handle expiry)
  • not handling 401: user stuck
  • trusting client: insecure
  • exposing token in URL: major leak
  • skipping backend: fake auth possible
  • not using HTTPS: insecure transport
